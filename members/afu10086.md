```java
 public static void main(String [] args){
        double sum = 0;
        for (int index = 1; index <= 100; index++) {
            double powIndex = Math.pow(3, 1.0 / index);
            double indexSum = index * index * index - powIndex;
            sum = sum + indexSum;
        }
        System.out.println("sum:" + sum);
    }
```
