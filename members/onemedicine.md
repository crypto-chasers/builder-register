package main

import (
	"fmt"
	"math"
)

func main() {
	var sum float64

	for n := range []int{100:0}{
		sum = sum + (math.Pow(float64(n), 3) - math.Cbrt(float64(n)))
	}
	fmt.Println(sum)	// 	// 2.5502149836096782e+07
}
