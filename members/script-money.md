```C#
"golang"
  
package main

import (
	"fmt"
	"math"
)

func main()  {
	var sum float64 = 0
	for i:= 1;i<=100;i++ {
		sum += math.Pow(float64(i),3) - math.Cbrt(float64(i))
	}
	fmt.Println(sum)
}
```
