```Golang
package main

import (
	"fmt"
	"math"
)

func main() {
	fmt.Println(calculate(100)) // 2.5502149836096782e+07
}

func calculate(n int) float64 {
	sum := float64(0)
	for i := 1; i <= n; i++ {
		sum += math.Pow(float64(i), 3) - math.Cbrt(float64(i))
	}
	return sum
}
```
